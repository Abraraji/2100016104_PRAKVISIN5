google.charts.load('current', {'packages':['corechart']}); google.charts.setOnLoadCallback(drawChart); function drawChart() { var queryString = encodeURIComponent('SELECT *'); var query = new google.visualization.Query('https://docs.google.com/spreadsheets/d/SPREADSHEET_ID/gviz/tq?tq=' + queryString); query.send(function(response) { if (response.isError()) { console.log('Terjadi kesalahan saat memuat data: ' + response.getMessage()); return; } var data = response.getDataTable(); var options = { title: 'Data Saya' }; var chart = new google.visualization.LineChart(document.getElementById('chart')); chart.draw(data, options); }); }
